var addStrings = function (num1, num2) {
  var i = num1.length - 1; j = num2.length - 1;
  var add = 0;
  var result = [];
  while (i >= 0 || j >= 0 || add !== 0) {
    //è¡¥ä½
    var x = i >= 0 ? num1[i] - '0' : 0;
    var y = j >= 0 ? num2[j] - '0' : 0;
    var temp = x + y + add;
    result.push(temp % 10);
    add = Math.floor(temp / 10);
    i -= 1;
    j -= 1;
  }
  return result.reverse().join('');
}
