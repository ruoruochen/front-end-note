### NC96 判断一个链表是否为回文结构
```js
function isPail( head ) {
    if(!head || !head.next) return true;
    let stack = [];
    let slow = head,fast = head;
    while(fast && fast.next){
        stack.push(slow.val);
        fast = fast.next.next;
        slow = slow.next;
    }
    if(fast){
        slow = slow.next;
    }
    while(slow){
        if(slow.val !== stack.pop()) return false;
        slow = slow.next;
    }
    return true;
}
```

自己不优雅的写法
```js
function isPail( head ) {
    if(!head || !head.next) return true;
   //计算长度
    let len = getListLength(head);
    //利用栈 存一半的链表
    let halflen = parseInt(len/2);
    let count =0;
    const stack = [];
    while(head){
        count++;
        if(count<=halflen){
            stack.push(head.val);
        }else{
            //中间位不参与比较
            if(len%2!==0){
                head = head.next;
                len = len-1;
                continue;
            }
            //比较
            if(stack.pop() !== head.val){
                return false;
            }
        }
        head = head.next;
    }
    return true;
}

function getListLength(head){
    let len = 0;
    while(head){
        len++;
        head = head.next;
    }
    return len;
}
```
