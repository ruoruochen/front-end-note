#### **NC96** **判断一个链表是否为回文结构**

方法1 变数组

```js
function isPail( head ) {
   const arr =[];
    while(head){
        arr.push(head);
        head = head.next;
    }
    let start = 0;
    let end = arr.length-1;
    while(start<=end){
        if(arr[start].val!==arr[end].val){
            return false;
        }
        start++;
        end--;
    }
    return true;
}
```
