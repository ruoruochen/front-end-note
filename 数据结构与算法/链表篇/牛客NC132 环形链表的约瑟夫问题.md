### NC132 环形链表的约瑟夫问题

#### 解法1
链表模拟环
```js
function ysf( n ,  m ) {
   //链表模拟环
    const head = {val:1};
    let cur = head;
    for(let i =2;i<=n;i++){
        cur.next = {val:i};
        cur = cur.next;
    }
    cur.next = head;
    //开始游戏
    while(cur.next !== cur){
        //报数 找到前一个人 跨过后一个
        for(let i =0;i<m-1;i++){
            cur = cur.next;
        }
        cur.next = cur.next.next;
    }
    return cur.val;
}
```

#### 解法2
数组模拟环
```js
function ysf( n ,  m ) {
    //数组模拟环
    let arr = [];
    for(let i =0;i<n;i++){
        arr[i] = i+1;
    }
    //开始游戏
    let index = 0; //当前报数人
    while(arr.length>1){
        //计算删除结点的下标
        index = (index+m)%arr.length -1;
        if(index>=0){
            arr.splice(index,1);
        }else{
            //删除末尾元素
            arr.splice(arr.length-1,1);
            index = 0;
        }
    }
    return arr[0];
}
```
