#### NC41 最长无重复子数组

```js
function maxLength( arr ) {
    let max = 1;
    let map = new Map();
    //左窗口边界
    let temp = 0;
    //滑动窗口
    //i为right边界 temp 为left边界
    //map 判断重复与否 键 数字 值 下标
    //无重复数组 = 重复数字的下一个-重复数字
    for(let i =0;i<arr.length;i++){
        if(map.has(arr[i])){
            //如果当前数字重复，找到重复的数字下标的下一个 窗口滑动
            temp = Math.max(temp,map.get(arr[i])+1);
        }
        //重新设计数字重复的下标
        map.set(arr[i],i);
        //重新获取无重复长度
        max = Math.max(max,i-temp+1);
    }
    return max;
}
```
